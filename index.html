2    then q ← PARTITION(A, p, r)

Procedure PARTITION(A, p, r)
1x ← A[r]
2i ← p - 1
3for j ← p to r - 1
4    do if A[j] ≤ x
5        then i ← i + 1
6             exchange A[i] ↔ A[j]
7exchange A[i + 1] ↔ A[r]
8return i + 1

Procedure HEAPSORT(A)
1BUILD_HEAP(A)
2for i ← length[A] down to 2
3    do exchange A[1] ↔ A[i]
4       heap_size[A] ← heap_size[A] - 1
5       HEAPIFY(A, 1)
Procedure BUILD_HEAP(A)
1heap_size[A] ← length[A]
2for i ← ⌊length[A] / 2⌋ downto 1
3    do HEAPIFY(A, i)
Procedure HEAPIFY(A, i)
1l ← LEFT_CHILD(i)
2r ← RIGHT_CHILD(i)
3if l ≤ heap_size[A] and A[l] > A[i]
4    then largest ← l
5    else largest ← i

8if largest ≠ i
